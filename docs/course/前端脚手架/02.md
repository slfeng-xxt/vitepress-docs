## 二、参数解析

```js
// 书接上文，fsl-test/bin/index.js
!#/user/bin/env node
const lib = require('fsl-test-lib')
console.log('test...lib', lib)
// 参数解析
const argv = require('process').argv
console.log(argv) // output: []
const commond = argv[2]
const options = argv.slice(3)
const [option, param] = options
option = option.replace('--', '')

if(commond) {
    if(lib[commond]){
        lib[commond]({option, param})
    } else {
        console.log('无效命令')
    }
} else {
    console.log('请输入命令')
}
```
